#include<iostream>
class Node{
    public:
    int val;
    Node*next;
    Node(int v){
        this->val=v; 
        this->next=NULL;
    }
};
int main(){
    Node*a=new Node(10);
    Node*b=new Node(20);
    Node*c=new Node(30);
    Node*d=new Node(40);
    a->next=b;
    b->next=c;
    c->next=d;
    Node*temp=a;
    while(temp!=NULL){
        std::cout<<temp->val<<" ";
        temp=temp->next;
    }
    
    std::cout<<(a->next)->next->next->val;
    
}
