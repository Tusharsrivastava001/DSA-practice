#include<iostream>
class volume{
  public:
  virtual void vol()=0;
};
class sphere:public volume{
    public:
    int r;
    sphere(int r){
        this->r=r;
    }
    void vol(){
        std::cout<<(4/3)*3.14*r*r*r;
    }
};
class cylinder:public volume{
    public:
    int r;
    int h;
    cylinder(int r,int h){
        this->r=r;
        this->h=h;
    }
    void vol(){
        std::cout<<3.14*r*r*h;
    }
};
int main(){
    sphere s(2);
    s.vol(); // Prints volume of the sphere
    std::cout<<std::endl;
    cylinder c(2, 5);
    c.vol();
    // Prints volume of the cylinder
