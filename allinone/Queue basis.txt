#include<iostream>
#include<queue>
void display(std::queue<int> &q){
    int n=q.size();
    for(int i=1;i<=n;i++){
        int x=q.front();
        std::cout<<x<<" ";
        q.pop();
        q.push(x);
    }
    
}
int main(){
    std::queue<int> q1;
    q1.push(10);
    q1.push(20);
    q1.push(30);
    q1.push(40);
    q1.push(50);
    display(q1);
    q1.pop();
    std::cout<<std::endl;
    display(q1);
 
    
    
}
