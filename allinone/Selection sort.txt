#include<iostream>
#include<climits>
int main(){
    int n;
    std::cout<<"Enter the size of array: ";
    std::cin>>n;
    int arr[n];
    for(int i=0;i<n;i++){
        std::cin>>arr[i];
    }
    
    for(int i=0;i<n-1;i++){ // last index nhi dekhna kyuki last me soretd ho chuka hoga
        int min=INT_MAX;
        int mindx=-1;
        for(int j=i;j<n;j++){ // travel in whole array
            if(arr[j]<min){
                min=arr[j];
                mindx=j;
            }
        }
        int temp=arr[i];
        arr[i]=arr[mindx];
        arr[mindx]=temp;
    }
    
    for(int i=0;i<n;i++){
        std::cout<<arr[i]<<" ";
    }
}
