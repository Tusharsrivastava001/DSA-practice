#include<iostream>
#include<stack>
int main(){
    std::stack<int> st;
    std::stack<int> temp;
    st.push(10);
    st.push(20);
    st.push(30);
    st.push(40);
    st.push(50);
    while(st.size()>0){
        std::cout<<st.top()<<" ";
        temp.push(st.top());
        st.pop();
    }
    temp.push(70);
    while(temp.size()>0){
        st.push(temp.top());
        temp.pop();
    }
    std::cout<<std::endl;
    while(st.size()>0){
       std::cout<<st.top()<<" ";
        temp.push(st.top());
        st.pop(); 
    }
}
