#include<iostream>
//array implentaion of stack
class Stack{
    public:
    int arr[5];
    int idx;
    Stack(){
        idx=-1;
    }
    void push(int val){
        if(idx==4){
            std::cout<<"Stack is full"<<std::endl;
            return;
        }
        idx++;
        arr[idx]=val;
    }
    void pop(){
        if(idx==-1){
            std::cout<<"Stack is empty"<<std::endl;
            return;
        }
        idx--;
    }
    int top(){
        return arr[idx];
    }
    int size(){
        return idx+1;
    }
};
int main(){
    
    Stack s1;
    s1.pop();
    s1.push(10);
    s1.push(20);
    s1.push(30);
    s1.push(40);
    s1.push(50);
    s1.push(60);
    std::cout<<s1.size()<<std::endl;

    
    
}
