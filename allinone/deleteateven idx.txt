#include<iostream>
#include<queue>
#include<stack>
void display(std::queue<int> &q){
    int n=q.size();
    for(int i=0;i<n;i++){
        int x=q.front();
        std::cout<<x<<" ";
        q.pop();
        q.push(x);
    }
    std::cout<<std::endl;
}
void removeateven(std::queue<int> &q,std::queue<int> &nw){
    int n=q.size();
    for(int i=0;i<n;i++){
        int x=q.front();
        if(i%2!=0){
            nw.push(x);
        }
        q.pop();
    }
}
void reverse(std::queue<int> &q){
    std::stack<int> a;
    int n=q.size();
    for(int i=0;i<n;i++){
        a.push(q.front());
        q.pop();
    }
    while(a.size()>0){
        q.push(a.top());
        a.pop();
    }
}
int main(){
    std::queue<int> s;
    std::queue<int> nw;
    s.push(10);
    s.push(20);
    s.push(30);
    s.push(40);
    s.push(50);
    s.push(60);
    display(s);
    removeateven(s,nw);
    display(nw);
    
}
