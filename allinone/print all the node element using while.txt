#include<iostream>
class Node{
    public:
    int val;
    Node*next;
    Node(int v){
        this->val=v;
        this->next=nullptr;
    }
};

int main(){
    Node a(10);
    Node b(20);
    Node c(30);
    Node d(40);
    a.next=&b;
    b.next=&c;
    c.next=&d;
    d.next=nullptr;
    Node temp=a;

    while(1){
        std::cout<<temp.val<<" ";
        if(temp.next==nullptr){
            break;
        }
        temp=(*(temp.next));
    }
    
}
