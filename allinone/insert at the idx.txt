#include<iostream>
#include<stack>
void print(std::stack<int> st){
    std::stack<int> temp;
    while(st.size()>0){
       temp.push(st.top());
       st.pop();
    }
    while(temp.size()>0){
        std::cout<<temp.top()<<" ";
        st.push(temp.top());
        temp.pop();
    }
}
void pushatbottom(std::stack<int> &st,int val){
    std::stack<int> temp;
    while(st.size()>0){
        temp.push(st.top());
        st.pop();
    }
    st.push(val);
    while(temp.size()>0){
        st.push(temp.top());
        temp.pop();
        
    }
}
void pushatidx(std::stack<int> &st,int val,int idx){
    std::stack<int> temp;
    while(st.size()>idx){
        temp.push(st.top());
        st.pop();
    }
    temp.push(val);
    while(st.size()>0){
        temp.push(st.top());
        st.pop();
    }
    while(temp.size()>0){
        st.push(temp.top());
        temp.pop();
    }
    
}
int main(){
    std::stack<int> st;
    st.push(10);
    st.push(20);
    st.push(30);
    st.push(40);
    st.push(50);
    print(st);
    std::cout<<std::endl;
    pushatbottom(st,80);
    print(st);
    std::cout<<std::endl;
    pushatidx(st,140,2);
    print(st);
    
}
