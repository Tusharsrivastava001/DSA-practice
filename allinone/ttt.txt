#include<iostream>
class abc{
    public:
    int a,b;
    abc(int a,int b){
        this->a=a;
        this->b=b;
    }
    friend void operator--(abc ob);
    void show(){
        std::cout<<a<<" "<<b<<std::endl;
    }
};
void operator--(abc ob){
    ob.a=ob.a-1;
    ob.b=ob.b-1;
    std::cout<<ob.a<<" "<<ob.b;
}
int main(){
    abc ob(12,34);
    --ob;

}
