#include<iostream>
class Node{
    public:
    int val;
    Node*next;
    Node(int val){
        this->val=val;
        this->next=NULL;
    }
};

int main(){
    Node*a=new Node(1);
    a->next=a;
    Node*temp=a;
    while(temp!=NULL){
        std::cout<<temp->val<<" ";
        temp=temp->next;
    }

}
