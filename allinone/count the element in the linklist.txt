#include<iostream>
class Node{
    public:
    int val;
    Node*next;
    Node(int v){
        this->val=v; 
        this->next=NULL;
    }
};
int size(Node*head){
   
    Node*temp=head;
     int n=0;
    while(temp!=NULL){
        n++;
        temp=temp->next;
    }
    return n;
}

int main(){
    Node*a=new Node(10);
    Node*b=new Node(20);
    Node*c=new Node(30);
    Node*d=new Node(40);
    Node*e=new Node(50);
    a->next=b;
    b->next=c;
    c->next=d;
    d->next=e;
    
    std::cout<<size(a);
}
